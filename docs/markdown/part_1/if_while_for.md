* –ï—Å–ª–∏ —É—Å–ª–æ–≤–∏–µ –∏—Å—Ç–∏–Ω–Ω–æ, —Ç–æ —Ä–∞–±–æ—Ç–∞–µ—Ç –≤—ã—Ä–∞–∂–µ–Ω–∏–µ –ø–æ–¥ –æ–ø–µ—Ä–∞—Ç–æ—Ä–æ–º `if`.
* –ï—Å–ª–∏ —É—Å–ª–æ–≤–∏–µ –≤ if –Ω–µ –≤—ã–ø–æ–ª–Ω—è–µ—Ç—Å—è, —Ç–æ –±—É–¥–µ—Ç –ø—Ä–æ–≤–µ—Ä–µ–Ω–æ –∞–ª—å—Ç–µ—Ä–Ω–∞—Ç–∏–≤–Ω–æ–µ —É—Å–ª–æ–≤–∏–µ –≤ `elif`.
* –í –ø—Ä–æ—Ç–∏–≤–Ω–æ–º —Å–ª—É—á–∞–µ, –µ—Å–ª–∏ –Ω–∏–∫–∞–∫–æ–µ –∏–∑ –ø—Ä–µ–¥–ª–æ–∂–µ–Ω–Ω—ã—Ö —É—Å–ª–æ–≤–∏–π –Ω–µ –≤—ã–ø–æ–ª–Ω—è–µ—Ç—Å—è, —Ç–æ —Ä–∞–±–æ—Ç–∞–µ—Ç –æ–ø–µ—Ä–∞—Ç–æ—Ä `else`.

```Python
if res > num:
    print(f'{res} –±–æ–ª—å—à–µ {num}')
elif res < l:
    print(f'{res} –º–µ–Ω—å—à–µ {num}')
else:
    print(f'{res} —Ä–∞–≤–µ–Ω {num}')
```

## –û–ø–µ—Ä–∞—Ç–æ—Ä—ã —Ü–∏–∫–ª–∞

–û–ø–µ—Ä–∞—Ç–æ—Ä—ã —Ü–∏–∫–ª–∞ –≤ Python –∏—Å–ø–æ–ª—å–∑—É—é—Ç—Å—è –¥–ª—è –≤—ã–ø–æ–ª–Ω–µ–Ω–∏—è –±–ª–æ–∫–∞ –∫–æ–¥–∞ –æ–ø—Ä–µ–¥–µ–ª–µ–Ω–Ω–æ–µ –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ —Ä–∞–∑ –∏–ª–∏ –¥–æ –≤—ã–ø–æ–ª–Ω–µ–Ω–∏—è –∑–∞–¥–∞–Ω–Ω–æ–≥–æ —É—Å–ª–æ–≤–∏—è. –û–Ω–∏ –Ω—É–∂–Ω—ã –¥–ª—è —Ç–æ–≥–æ, —á—Ç–æ–±—ã –≤—ã–ø–æ–ª–Ω—è—Ç—å –æ–¥–Ω–æ—Ç–∏–ø–Ω—ã–µ –¥–µ–π—Å—Ç–≤–∏—è –º–Ω–æ–≥–æ–∫—Ä–∞—Ç–Ω–æ. 

–í–æ—Ç –Ω–∞–∏–±–æ–ª–µ–µ —Ä–∞—Å–ø—Ä–æ—Å—Ç—Ä–∞–Ω–µ–Ω–Ω—ã–µ –≤–∏–¥—ã —Ü–∏–∫–ª–æ–≤ –≤ Python:

* `for` –∏—Å–ø–æ–ª—å–∑—É–µ—Ç—Å—è –¥–ª—è –∏—Ç–µ—Ä–∞—Ü–∏–∏ –ø–æ –∫–∞–∂–¥–æ–º—É —ç–ª–µ–º–µ–Ω—Ç—É –≤ –∫–æ–ª–ª–µ–∫—Ü–∏–∏.
* `while` –≤—ã–ø–æ–ª–Ω—è–µ—Ç—Å—è –ø–æ–∫–∞ –∑–∞–¥–∞–Ω–Ω–æ–µ —É—Å–ª–æ–≤–∏–µ –æ—Å—Ç–∞–µ—Ç—Å—è –∏—Å—Ç–∏–Ω–Ω—ã–º.

### –û–ø–µ—Ä–∞—Ç–æ—Ä while

–û–ø–µ—Ä–∞—Ç–æ—Ä `while` –ø–æ–∑–≤–æ–ª—è–µ—Ç –º–Ω–æ–≥–æ–∫—Ä–∞—Ç–Ω–æ –≤—ã–ø–æ–ª–Ω—è—Ç—å –∫–æ–¥ –≤ –∑–∞–≤–∏—Å–∏–º–æ—Å—Ç–∏ –æ—Ç –∑–∞–¥–∞–Ω–Ω–æ–≥–æ –ª–æ–≥–∏—á–µ—Å–∫–æ–≥–æ —É—Å–ª–æ–≤–∏—è.

–†–∞—Å—Å–º–æ—Ç—Ä–∏–º –ø—Ä–æ—Å—Ç–æ–π –ø—Ä–∏–º–µ—Ä. –ü—Ä–µ–¥—Å—Ç–∞–≤–∏–º, —á—Ç–æ –Ω–∞–º –Ω–µ–æ–±—Ö–æ–¥–∏–º–æ –≤—ã–≤–µ—Å—Ç–∏ –Ω–∞ —ç–∫—Ä–∞–Ω —Ç–µ–∫—É—â–µ–µ –∑–Ω–∞—á–µ–Ω–∏–µ —Å—á–µ—Ç—á–∏–∫–∞ `i`, –ø–æ–∫–∞ –∑–Ω–∞—á–µ–Ω–∏–µ –Ω–µ –±—É–¥–µ—Ç –±–æ–ª—å—à–µ 10. –ß—Ç–æ–±—ã –Ω–µ –ø–∏—Å–∞—Ç—å –∫–∞–∂–¥—ã–π —Ä–∞–∑ —Ñ—É–Ω–∫—Ü–∏—é `print()` —Å –Ω–æ–≤—ã–º –∑–Ω–∞—á–µ–Ω–∏–µ–º `i`, –º–æ–∂–Ω–æ —É–ø—Ä–æ—Å—Ç–∏—Ç—å –∫–æ–¥ –∏ –∏—Å–ø–æ–ª—å–∑–æ–≤–∞—Ç—å —Ü–∏–∫–ª `while`, –∑–∞–ø–∏—Å–∞–≤ —É—Å–ª–æ–≤–∏–µ **"–ø–æ–∫–∞ `i` –º–µ–Ω—å—à–µ –ª–∏–±–æ —Ä–∞–≤–µ–Ω 10"** –∏ —É–∫–∞–∑–∞–≤ –æ–ø–µ—Ä–∞—Ü–∏—è –ø–æ–¥ —ç—Ç–∏–º —É—Å–ª–æ–≤–∏–µ–º.

```Python
i = 1                  # i —Ö—Ä–∞–Ω–∏—Ç 1 (int)

    print('i =', i)        # –≤—ã–≤–æ–¥–∏—Ç—å –ø–æ–ª—É—á–µ–Ω–Ω—ã–π i.

print('–î–æ —Å–≤–∏–¥–∞–Ω–∏—è!')  # –¥–æ —Å–≤–∏–¥–∞–Ω–∏—è!
```

**–î–ª—è —á–µ–≥–æ –Ω—É–∂–Ω–∞ –æ–ø–µ—Ä–∞—Ü–∏—è `i = i + 1`?**

–†–∞—Å—Å–º–æ—Ç—Ä–∏–º –ø–æ —à–∞–≥–∞–º, –∫–∞–∫ –±–ª–æ–∫ –∫–æ–¥–∞ –≤—ã—à–µ –±—É–¥–µ—Ç –≤—ã–ø–æ–ª–Ω—è—Ç—Å—è –±–µ–∑ —ç—Ç–æ–π —Å—Ç—Ä–æ–∫–∏.

1. `i` –ø—Ä–∏–Ω—è–ª –∑–Ω–∞—á–µ–Ω–∏–µ 1.
2. –ø—Ä–æ–≤–µ—Ä—è–µ–º —É—Å–ª–æ–≤–∏–µ —É while: `i <= 10` -> **–ò—Å—Ç–∏–Ω–∞**.
3. –≤—Ö–æ–¥–∏–º –≤ —É—Å–ª–æ–≤–∏–µ –∏ –≤—ã–ø–æ–ª–Ω—è–µ–º —Ñ—É–Ω–∫—Ü–∏—é `print('i =', i)`.
4. –≤ –∫–æ–Ω—Å–æ–ª–∏ –ø–æ—è–≤–ª—è–µ—Ç—Å—è —Å–æ–æ–±—â–µ–Ω–∏–µ `i = 1`.
5. –≤–æ–∑–≤—Ä–∞—â–∞–µ–º—Å—è –≤ —É—Å–ª–æ–≤–∏–µ —Ü–∏–∫–ª–∞ while `i <= 10`
6. **—Ç–∞–∫ –∫–∞–∫ `i = 1`, —Ç–æ –≤ —É—Å–ª–æ–≤–∏–∏ –æ–ø—è—Ç—å –±—É–¥–µ—Ç –ø—Ä–æ–≤–µ—Ä–µ–Ω —Ç–æ—Ç –∂–µ `i`. –ò —Ç–∞–∫ –±—É–¥–µ—Ç –ø—Ä–æ–¥–æ–ª–∂–∞—Ç—å—Å—è –≤–µ—á–Ω–æ!**

```Python
i = 1

    print('i =', i)

# –û–®–ò–ë–ö–ê –í–ï–ß–ù–û–ì–û –¶–ò–ö–õ–ê!
# –ß—Ç–æ–±—ã –æ—Å—Ç–∞–Ω–æ–≤–∏—Ç—å —Ä–∞–±–æ—Ç—É –∏–Ω—Ç–µ—Ä–ø—Ä–µ—Ç–∞—Ç–æ—Ä–∞,
# –Ω–∞–∂–º–∏—Ç–µ Ctrl + C
```

–ü–æ—ç—Ç–æ–º—É, –µ—Å–ª–∏ –≤—ã –∏—Å–ø–æ–ª—å–∑—É–µ—Ç–µ —Ü–∏–∫–ª —Å `while`, **–Ω–µ –∑–∞–±—ã–≤–∞–π—Ç–µ –¥–µ–ª–∞—Ç—å —à–∞–≥ –≤ —Å—á–µ—Ç—á–∏–∫–µ, –æ—Ç –∫–æ—Ç–æ—Ä–æ–≥–æ –∑–∞–≤–∏—Å–∏—Ç –¥–≤–∏–∂–µ–Ω–∏–µ —Ü–∏–∫–ª–∞**!

–ü—Ä–∏–±–∞–≤–ª–µ–Ω–∏–µ –µ–¥–∏–Ω–∏—Ü—ã –∫ —Å–∞–º–æ–π –ø–µ—Ä–µ–º–µ–Ω–Ω–æ–π `i` –¥–æ–≤–æ–ª—å–Ω–æ —á–∞—Å—Ç–∞—è –æ–ø–µ—Ä–∞—Ü–∏—è. –ü–æ—ç—Ç–æ–º—É —Å—É—â–µ—Å—Ç–≤—É—é—Ç —Å–æ–∫—Ä–∞—â–µ–Ω–Ω—ã–µ –∑–∞–ø–∏—Å–∏ —Ç–∞–∫–æ–π –æ–ø–µ—Ä–∞—Ü–∏–∏, –∫–æ—Ç–æ—Ä–∞—è –∑–∞–ø–∏—Å—ã–≤–∞–µ—Ç—Å—è —Å –ø–æ–º–æ—â—å—é –æ–ø–µ—Ä–∞—Ç–æ—Ä–∞ `+=`. –¢–∞–∫–∂–µ —Å—É—â–µ—Å—Ç–≤—É—é—Ç `-=` –∏ `*=`.

```Python
i = 1

    i += 1  # —Ç–æ–∂–µ —Å–∞–º–æ–µ, —á—Ç–æ i = i + 1
    print('i =', i)

```

–û–ø–µ—Ä–∞—Ç–æ—Ä `if` —Ç–∞–∫–∂–µ –º–æ–∂–Ω–æ –∑–∞–ø–∏—Å—ã–≤–∞—Ç—å –≤–Ω—É—Ç—Ä–∏ —Ü–∏–∫–ª–æ–≤. –†–∞—Å—Å–º–æ—Ç—Ä–∏–º –ø—Ä–∏–º–µ—Ä. –ü–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—å –≤–≤–æ–¥–∏—Ç —Å –∫–æ–Ω—Å–æ–ª–∏ —á–∏—Å–ª–æ —Å –ø–æ–º–æ—â—å—é —Ñ—É–Ω–∫—Ü–∏–∏ `input()`. –ï—Å–ª–∏ –æ–Ω –ø–æ–ø–∞–¥–∞–µ—Ç –≤ –∏–Ω—Ç–µ—Ä–≤–∞–ª –æ—Ç 1 –¥–æ 5, —Ç–æ –≤ –∫–æ–Ω—Å–æ–ª–∏ –≤—ã–≤–æ–¥–∏—Ç—Å—è —Å–æ–æ–±—â–µ–Ω–∏–µ _'–í—ã –≤—ã–∏–≥—Ä–∞–ª–∏!'_.

```Python
i = 0

while i < 5:
    if i == user_input:
        print('–í—ã –≤—ã–∏–≥—Ä–∞–ª–∏!')
    print(i)
    i += 1 
```

–û–±—Ä–∞—Ç–∏—Ç–µ –≤–Ω–∏–º–∞–Ω–∏–µ –Ω–∞ –æ–ø–µ—Ä–∞—Ü–∏—é –ø—Ä–∏–≤–µ–¥–µ–Ω–∏—è –∫ —Ç–∏–ø—É `int()` –ø–µ—Ä–µ–¥ input, —Ç–∞–∫ –∫–∞–∫ —Ä–µ–∑—É–ª—å—Ç–∞—Ç —Ñ—É–Ω–∫—Ü–∏–∏ input - —ç—Ç–æ `str`, –∞ **—Å—Ä–∞–≤–Ω–∏–≤–∞—Ç—å `int` –º–æ–∂–Ω–æ —Ç–æ–ª—å–∫–æ —Å `int`**.

–í –ø—Ä–∏–º–µ—Ä–µ –≤—ã—à–µ –≤—ã–ø–æ–ª–Ω–µ–Ω–∏–µ —Ü–∏–∫–ª–∞ –Ω–µ –æ—Å—Ç–∞–Ω–∞–≤–ª–∏–≤–∞–µ—Ç—Å—è, –∫–æ–≥–¥–∞ –≤—ã–ø–æ–ª–Ω—è–µ—Ç—Å—è —É—Å–ª–æ–≤–∏–µ. –û–ø–µ—Ä–∞—Ç–æ—Ä `break` –ø–æ–∑–≤–æ–ª—è–µ—Ç –ø—Ä–µ—Ä–≤–∞—Ç—å —Ü–∏–∫–ª—ã, –∫–æ–≥–¥–∞ —É—Å–ª–æ–≤–∏–µ –≤—ã–ø–æ–ª–Ω–∏—Ç—Å—è. **break –º–æ–∂–Ω–æ –∏—Å–ø–æ–ª—å–∑–æ–≤–∞—Ç—å —Ç–æ–ª—å–∫–æ –≤–Ω—É—Ç—Ä–∏ —Ü–∏–∫–ª–æ–≤!**

```Python
i = 0

while i < 5:
    if i == user_input:
        print('–í—ã –≤—ã–∏–≥—Ä–∞–ª–∏!')
        break  # break - –ø—Ä–µ–∫—Ä–∞—â–∞–µ—Ç —Ü–∏–∫–ª
    print(i)
    i += 1
```

–û–ø–µ—Ä–∞—Ç–æ—Ä—ã –∏–Ω—Å—Ç—Ä—É–∫—Ü–∏–∏ `if-elif-else` –º–æ–∂–Ω–æ –≤–≤–æ–¥–∏—Ç—å –Ω–∞ —Ä–∞–∑–Ω—ã—Ö —É—Ä–æ–≤–Ω—è—Ö. –ù–∞–ø—Ä–∏–º–µ—Ä, –∫–æ–≥–¥–∞ —Ü–∏–∫–ª –∑–∞–≤–µ—Ä—à–∏—Ç —Ä–∞–±–æ—Ç—É –∏ —É—Å–ª–æ–≤–∏–µ `if` –≤–Ω—É—Ç—Ä–∏ –Ω–µ –±—É–¥–µ—Ç –≤—ã–ø–æ–ª–Ω–µ–Ω–æ, —Ç–æ **–ø–æ—Å–ª–µ —Ü–∏–∫–ª–∞ —Å—Ä–∞–±–æ—Ç–∞–µ—Ç –æ–ø–µ—Ä–∞—Ç–æ—Ä `else`**.

```Python
i = 0

        print('–í—ã –≤—ã–∏–≥—Ä–∞–ª–∏!')
```

### –û–ø–µ—Ä–∞—Ç–æ—Ä for

–ó–∞—á–∞—Å—Ç—É—é –Ω–µ–æ–±—Ö–æ–¥–∏–º–æ –≤—ã–ø–æ–ª–Ω—è—Ç—å –æ–ø–µ—Ä–∞—Ü–∏–∏ –Ω–∞–¥ —ç–ª–µ–º–µ–Ω—Ç–∞–º–∏ –∫–æ–ª–ª–µ–∫—Ü–∏–∏. –ß—Ç–æ–±—ã –ø–æ–ª—É—á–∏—Ç—å –∫–∞–∂–¥—ã–π —ç–ª–µ–º–µ–Ω—Ç –ø–æ—Å–ª–µ–¥–æ–≤–∞—Ç–µ–ª—å–Ω–æ, –Ω—É–∂–Ω–æ –∑–Ω–∞—Ç—å –∏–Ω–¥–µ–∫—Å –∫–∞–∂–¥–æ–≥–æ —ç–ª–µ–º–µ–Ω—Ç–∞ –∫–æ–ª–ª–µ–∫—Ü–∏–∏. –¢–∞–∫—É—é –∑–∞–¥–∞—á—É –º–æ–∂–Ω–æ –≤—ã–ø–æ–ª–Ω–∏—Ç—å —Å –ø–æ–º–æ—â—å—é —Ü–∏–∫–ª–∞ `while` –≤ —Å–ª–µ–¥—É—é—â–µ–º –ø—Ä–∏–º–µ—Ä–µ.

–ß—Ç–æ–±—ã —É–∑–Ω–∞—Ç—å –¥–ª–∏–Ω—É —Å—Ç—Ä–æ–∫–∏, —Å–ø–∏—Å–∫–∞ –∏–ª–∏ –∫–æ—Ä—Ç–µ–∂–∞, –º–æ–∂–Ω–æ –≤–æ—Å–ø–æ–ª—å–∑–æ–≤–∞—Ç—å—Å—è —Ñ—É–Ω–∫—Ü–∏–µ–π `len()`.

```Python
# –í–µ—Ä–Ω–µ–º—Å—è –∫ –∏—Å—Ö–æ–¥–Ω–æ–π —á–∏—Å–ª–æ–≤–æ–π –ø–æ—Å–ª–µ–¥–æ–≤–∞—Ç–µ–ª—å–Ω–æ—Å—Ç–∏
primes = range(10, 100, 15)
i = 0  # –ø–µ—Ä–≤—ã–π –∏–Ω–¥–µ–∫—Å 0

while i < len(primes): # –ø–æ–∫–∞ –∏–Ω–¥–µ–∫—Å i –º–µ–Ω—å—à–µ –¥–ª–∏–Ω—ã primes
    print(primes[i])  # –≤—ã–≤–æ–¥ —ç–ª–µ–º–µ–Ω—Ç–∞ —Å–ø–∏—Å–∫–∞ –ø–æ –∏–Ω–¥–µ–∫—Å—É
    i += 1
```

–ï—Å—Ç—å —Ä—è–¥ —Ç—Ä—É–¥–Ω–æ—Å—Ç–µ–π —Å –ø—Ä–∏–º–µ–Ω–µ–Ω–∏–µ–º —Ü–∏–∫–ª–∞ `while` –¥–ª—è —Å–ø–∏—Å–∫–æ–≤:

1. –Ω—É–∂–Ω–æ —á–µ—Ç–∫–æ –æ—Ç—Å–ª–µ–∂–∏–≤–∞—Ç—å –∏–∑–º–µ–Ω–µ–Ω–∏—è —Ç–µ–∫—É—â–µ–≥–æ –∏–Ω–¥–µ–∫—Å–∞,
2. –∏–Ω–¥–µ–∫—Å –Ω–µ –¥–æ–ª–∂–µ–Ω –ø–µ—Ä–µ–π—Ç–∏ –¥–ª–∏–Ω—É —Å–ø–∏—Å–∫–∞ –≤–æ –∏–∑–±–µ–∂–∞–Ω–∏–∏ –æ—à–∏–±–∫–∏ `Index out of range`.

–û–ø–µ—Ä–∞—Ç–æ—Ä `for` –ø–æ–∑–≤–æ–ª—è–µ—Ç –ø–µ—Ä–µ–±–∏—Ä–∞—Ç—å –≤—Å–µ —ç–ª–µ–º–µ–Ω—Ç—ã —Å—Ç—Ä–æ–∫, —Å–ø–∏—Å–∫–æ–≤, –∫–æ—Ä—Ç–µ–∂–µ–π –∏ –¥—Ä—É–≥–∏—Ö –∫–æ–ª–ª–µ–∫—Ü–∏–π —Å –ø–æ–º–æ—â—å—é –∑–∞–¥–∞–Ω–Ω–æ–π –ø–µ—Ä–µ–º–µ–Ω–Ω–æ–π.

```Python
primes = range(10, 100, 15)

for i in primes:  # –¥–ª—è –∫–∞–∂–¥–æ–≥–æ i –∏–∑ primes
    print(i)
```

–û–±—Ä–∞—Ç–∏—Ç–µ –≤–Ω–∏–º–∞–Ω–∏–µ, —á—Ç–æ –≤ –¥–∞–Ω–Ω–æ–º —Å–ª—É—á–∞–µ **`i` –±—É–¥–µ—Ç —Ö—Ä–∞–Ω–∏—Ç—å —ç–ª–µ–º–µ–Ω—Ç —Å–ø–∏—Å–∫–∞, –∞ –Ω–µ –∏–Ω–¥–µ–∫—Å**.

–†–∞—Å—Å–º–æ—Ç—Ä–∏–º –ø—Ä–∏–º–µ–Ω–µ–Ω–∏–µ —Ü–∏–∫–ª–∞ `for` –¥–ª—è —Å—Ç—Ä–æ–∫–∏. –ù–µ–æ–±—Ö–æ–¥–∏–º–æ –≤—ã—Ç–∞—â–∏—Ç—å –≤—Å–µ —Ü–∏—Ñ—Ä—ã –∏–∑ –∑–∞–¥–∞–Ω–Ω–æ–π —Å—Ç—Ä–æ–∫–∏. –ö–∞–∂–¥—ã–π —ç–ª–µ–º–µ–Ω—Ç –º–æ–∂–Ω–æ –ø—Ä–æ–≤–µ—Ä–∏—Ç—å —Å –ø–æ–º–æ—â—å—é –º–µ—Ç–æ–¥–∞ —Å—Ç—Ä–æ–∫–∏ `isdigit()`, —è–≤–ª—è–µ—Ç—Å—è –ª–∏ –æ–Ω —á–∏—Å–ª–æ–º.

–û–ø–µ—Ä–∞—Ç–æ—Ä `pass` –ø—Ä–æ–ø—É—Å–∫–∞–µ—Ç –∫–∞–∫–∏–µ-–ª–∏–±–æ –¥–µ–π—Å—Ç–≤–∏—è. –¢–∞–∫–∂–µ –µ—â–µ —Å—É—â–µ—Å—Ç–≤—É–µ—Ç –æ–ø–µ—Ä–∞—Ç–æ—Ä continue, –æ –ø—Ä–∏–º–µ–Ω–µ–Ω–∏–∏ –∫–æ—Ç–æ—Ä–æ–≥–æ –º–æ–∂–Ω–æ –ø—Ä–æ—á–∏—Ç–∞—Ç—å [–ø–æ —Å—Å—ã–ª–∫–µ](https://www.digitalocean.com/community/tutorials/how-to-use-break-continue-and-pass-statements-when-working-with-loops-in-python-3).

```Python

    if i.isdigit() == True: # —è–≤–ª—è–µ—Ç—Å—è –ª–∏ —Å–∏–º–≤–æ–ª —Ü–∏—Ñ—Ä–æ–π
        new_s += i # –¥–æ–±–∞–≤–∏—Ç—å —Å–∏–º–≤–æ–ª —Å—Ç—Ä–æ–∫—É
    else:
        pass # –ø—Ä–æ–ø—É—Å—Ç–∏—Ç—å —Å–∏–º–≤–æ–ª
```

```Python

# –í–∞—Ä–∏–∞–Ω—Ç 3 - –≤–æ–∑–≤—Ä–∞—Ç –∏–Ω–¥–µ–∫—Å–æ–≤ –∏ —ç–ª–µ–º–µ–Ω—Ç–æ–≤
```

### –ò–Ω—Å—Ç—Ä—É–∫—Ü–∏—è try-except

–í Python –∏–Ω—Å—Ç—Ä—É–∫—Ü–∏—è `try-except` –∏—Å–ø–æ–ª—å–∑—É–µ—Ç—Å—è –¥–ª—è –æ–±—Ä–∞–±–æ—Ç–∫–∏ –æ—à–∏–±–æ–∫ –≤ –∫–æ–¥–µ. –û–Ω–∞ –ø–æ–∑–≤–æ–ª—è–µ—Ç –≤—ã–ø–æ–ª–Ω–∏—Ç—å –±–ª–æ–∫ –∫–æ–¥–∞, –∫–æ—Ç–æ—Ä—ã–π –º–æ–∂–µ—Ç –≤—ã–∑–≤–∞—Ç—å –∏—Å–∫–ª—é—á–µ–Ω–∏–µ, –∏ –∏–∑–±–µ–∂–∞—Ç—å –∞–≤–∞—Ä–∏–π–Ω–æ–≥–æ –∑–∞–≤–µ—Ä—à–µ–Ω–∏—è –ø—Ä–æ–≥—Ä–∞–º–º—ã.  

**–ò—Å–∫–ª—é—á–µ–Ω–∏—è** ‚Äî —ç—Ç–æ –æ—à–∏–±–∫–∏, –∫–æ—Ç–æ—Ä—ã–µ –º–æ–≥—É—Ç –≤–æ–∑–Ω–∏–∫–Ω—É—Ç—å –≤–æ –≤—Ä–µ–º—è –≤—ã–ø–æ–ª–Ω–µ–Ω–∏—è –ø—Ä–æ–≥—Ä–∞–º–º—ã. –û–Ω–∏ –º–æ–≥—É—Ç –±—ã—Ç—å —Ä–µ–∑—É–ª—å—Ç–∞—Ç–æ–º –¥–µ–ª–µ–Ω–∏—è –Ω–∞ –Ω–æ–ª—å, –æ–±—Ä–∞—â–µ–Ω–∏—è –∫ –Ω–µ—Å—É—â–µ—Å—Ç–≤—É—é—â–µ–º—É —ç–ª–µ–º–µ–Ω—Ç—É —Å–ø–∏—Å–∫–∞ –∏ —Ç.–¥.

```Python
x = 5

print(f"–†–µ–∑—É–ª—å—Ç–∞—Ç: {x/y}") # –û—à–∏–±–∫–∞ ZeroDivisionError
```

–í –ø—Ä–∏–º–µ—Ä–µ –Ω–∏–∂–µ –ø–æ–∫–∞–∑–∞–Ω –ø—Ä–∏–º–µ—Ä, –≤ –∫–æ—Ç–æ—Ä–æ–º –Ω–µ–æ–±—Ö–æ–¥–∏–º–æ –æ–±—Ä–∞–±–æ—Ç–∞—Ç—å –æ—à–∏–±–∫—É –¥–µ–ª–µ–Ω–∏—è –Ω–∞ –Ω–æ–ª—å.

```Python
x = 5
y = 0

try:  # —Å–Ω–∞—á–∞–ª–∞ –∫–æ–¥ –≤—ã–ø–æ–ª–Ω—è–µ—Ç—Å—è –∑–¥–µ—Å—å
    print(f"–†–µ–∑—É–ª—å—Ç–∞—Ç: {x/y}")
except ZeroDivisionError: # –µ—Å–ª–∏ –≤ try –æ—à–∏–±–∫–∞, —Ç–æ –∫–æ–¥ –≤—ã–ø–æ–ª–Ω—è–µ—Ç—Å—è –∑–¥–µ—Å—å
    print('–ù–∞ 0 –¥–µ–ª–∏—Ç—å –Ω–µ–ª—å–∑—è!')
```

–û–ø–µ—Ä–∞—Ç–æ—Ä—ã `try` –∏ `except` —è–≤–ª—è—é—Ç—Å—è –æ–±—è–∑–∞—Ç–µ–ª—å–Ω—ã–º–∏ —á–∞—Å—Ç—è–º–∏ –∏–Ω—Å—Ç—Ä—É–∫—Ü–∏–∏, `else` –∏ `finally` - –æ–ø—Ü–∏–æ–Ω–∞–ª—å–Ω—ã–º–∏.

```Python

    print(f"–†–µ–∑—É–ª—å—Ç–∞—Ç: {x/y}")

    print('–ù–∞ 0 –¥–µ–ª–∏—Ç—å –Ω–µ–ª—å–∑—è!')

    print(f"–°–ª–µ–¥—É—é—â–∏–π —Ä–µ–∑—É–ª—å—Ç–∞—Ç: {x*2}")
finally: # –≤—ã–ø–æ–ª–Ω—è–µ—Ç—Å—è –≤ –ª—é–±–æ–º —Å–ª—É—á–∞–µ, –Ω–µ–∑–∞–≤–∏—Å–∏–º–æ –æ—Ç—Ä–∞–±–æ—Ç–∞–ª –ª–∏ –±–ª–æ–∫ try –∏–ª–∏ except
    print(f"–†–µ–∑—É–ª—å—Ç–∞—Ç: {x}")
```

–ü–æ–¥—Ä–æ–±–Ω–µ–µ –ø—Ä–æ –ø—Ä–∏–º–µ–Ω–µ–Ω–∏–µ try-except –º—ã —Ä–∞—Å—Å–º–æ—Ç—Ä–∏–º –≤–æ –≤—Ç–æ—Ä–æ–π —á–∞—Å—Ç–∏ –∫—É—Ä—Å–∞.

### –ò–Ω—Å—Ç—Ä—É–∫—Ü–∏—è match-case

–í Python –æ–ø–µ—Ä–∞—Ç–æ—Ä `match-case` –∏—Å–ø–æ–ª—å–∑—É–µ—Ç—Å—è –¥–ª—è –≤—ã–±–æ—Ä–∞ –æ–¥–Ω–æ–≥–æ –∏–∑ –Ω–µ—Å–∫–æ–ª—å–∫–∏—Ö –±–ª–æ–∫–æ–≤ –∫–æ–¥–∞ –¥–ª—è –≤—ã–ø–æ–ª–Ω–µ–Ω–∏—è –≤ –∑–∞–≤–∏—Å–∏–º–æ—Å—Ç–∏ –æ—Ç —É—Å–ª–æ–≤–∏—è. –ò–Ω—Å—Ç—Ä—É–∫—Ü–∏—è –∞–Ω–∞–ª–æ–≥–∏—á–Ω–∞ `if-elif-else` - –ø–æ–∑–≤–æ–ª—è–µ—Ç —Ä–∞–±–æ—Ç–∞—Ç—å —Å–æ —Å–ª–æ–∂–Ω—ã–º–∏ —É—Å–ª–æ–≤–∏—è–º–∏, –Ω–æ –∏–º–µ–µ—Ç –±–æ–ª–µ–µ —Å—Ç—Ä—É–∫—Ç—É—Ä–∏—Ä–æ–≤–∞–Ω–Ω—ã–π –∏ —á–∏—Ç–∞–µ–º—ã–π –≤–∏–¥.
